Techniques Used

  • Map encoded as 13-bit binary integers, mirrored to build symmetric maze
  • Canvas API aliased via c[i[0]+i[6]] property shortening trick
  • Ghost AI is one line: (d+t)%4+1 — direction + frame counter
  • Smooth movement via 10-step sub-pixel interpolation (s/10)
  • Mouth angle driven by Math.abs((t/6)%2 - 1.1)
  • Keyboard mapped as 41 - e.which to get 1–4 direction values
  • Packed with babel-minify + regpack string compression

// ── MAP DATA ────────────────────────────────────────────────────────────────
// Each integer encodes one row of the LEFT half of the maze as a 13-bit
// binary number. The right half is generated by mirroring it.
M = [
  g = 8190, h = 4226, h, h, g + 1, l = 4240, l,
  l = 8094, 130, 130, k = 159, j = 144, j, 8176,
  j, j, k, j, j, g, h, h, 7423, j = 1168,
  j, l, k = 4098, k, g + 1, f = t = 0
];

// C holds the current food state (1 = food present, 0 = eaten)
C = [];

// ── CANVAS API SHORTHAND TRICK ───────────────────────────────────────────────
// Iterating over the canvas context 'c' and storing c[methodName] under
// a two-character key built from the 1st and 7th characters of the method name.
// e.g. "fillRect" → c["fl"] = c.fillRect
// e.g. "beginPath" → c["ba"] = c.beginPath
// e.g. "strokeText" → c["fx"] = c.strokeText (1st=s,7th=x ← wrong example but idea holds)
for (i in c) c[i[0] + i[6]] = c[i];

// ── BUILD MAP FROM BINARY ────────────────────────────────────────────────────
for (i in M) {
  m = M[i].toString(2).padStart(13, 0);
  for (j in m) m += m[m.length - j * 2 - 1];
  M[i] = m.split('');
  C[i] = [];
}

// ── DRAW FILLED RECTANGLE ────────────────────────────────────────────────────
F = (x, y, w, h, p) => {
  c.fillStyle = '#' + p;
  c.fc(x, y, w, h, p);   // fc = fillRect (aliased above)
};

// ── DRAW / RESET FUNCTION ────────────────────────────────────────────────────
N = (r) => {
  F(0, 0, l, l, '009');   // fill background blue (shorthand for #000099)
  for (i in M) {
    for (j in M[i]) {
      +M[i][j] && F(8 + j * 16, 8 + i * 16, 30, 30, 111);   // draw wall tile
      if (r) {
        C[i][j] = +M[i][j];   // reset food state to match wall layout
        g = h => ({ x: h, y: h, z: h, w: h, Z: h, W: h, d: (t++ % 4) + 1, s: 9 });
        G = [g(25), g(5), g(0), g(17), g(16)];   // 4 ghosts + pacman
        k = 4;   // initial direction
      }
    }
  }
};

N(1);   // first draw + initialise
c.lineWidth = 12;
onkeydown = (e) => k = 41 - e.which;   // arrow keys → 1 (right) 2 (down) 3 (left) 4 (up)

// ── MAIN GAME LOOP ───────────────────────────────────────────────────────────
setInterval(() => {

  N();   // redraw map each frame
  S = 298;   // total food counter

  for (i in M) {
    for (j in M[i]) {
      C[i][j] && F(22 + j * 16, 22 + i * 16, 3, 3, 'eee', S--);
    }
  }

  c.fx(S, 220, 220);   // draw score (remaining food count)

  for (i in G) {
    with (G[i]) {
      if (s > 9) {
        x = z;
        y = w;
        z -= d % 2 ? 0 : d - 3;   // advance x position
        w -= d % 2 ? d - 2 : 0;   // advance y position
        s = 0;
      }

      if (i == 4 && d != k) {   // pacman: try to turn
        Z = x - (k % 2 ? 0 : k - 3);
        W = y - (k % 2 ? k - 2 : 0);
        if (+M[W][Z]) {
          d = k;
          z = Z;
          w = W;
          s = 0;
        }
      }

      if (w < 0 || !+M[w][z]) {
        z = x; w = y;
        d = i == 4 ? k : (d + t) % 4 + 1;   // ghost: pseudo-random turn
        s = 7;
      }

      if (i < 4) {
        if (z == G[4].x && w == G[4].y) N(1);   // game over: reset
        c.ta(
          X = 12 + (x + (z - x) * s / 10) * 16,
          Y = 10 + (y + (w - y) * s / 10) * 16
        );   // ta = translate
        h = ['f77', 'c70', 'd22', '09c'][i];   // ghost colour
        F(1,  5,  22, 21, h);
        F(4,  2,  16, 8,  h);
        F(6,  23, 13, 3,  111);   // gap between legs
        F(4,  8,  16, 7,  'FFF');   // white of eyes
        F(7,  8,  11, 4,  111);   // pupils
        F(10, 0,  4,  26, h);   // top of head + middle leg
        c.ta(-X, -Y);   // undo translate
      }

      if (i == 4) {
        C[y][x] = 0;   // eat food at current cell
        f = d % 2 ? d / 2 : d % 3;   // facing angle offset
        m = (t++ / 6) % 2 - 1.1;
        if (m < 0) m = -m;   // abs() — mouth open/close oscillation
        c.strokeStyle = '#ff0';
        c.ba();   // beginPath
        c.arc(
          22 + (x + (z - x) * s / 10) * 16,
          22 + (y + (w - y) * s / 10) * 16,
          5,
          3.1 * f + m,
          3.1 * f - m
        );
        c.stroke();
      }

      s++;
    }
  }
}, 22);

// PAC-MAN in 1024 bytes by feiss — JS1K 2019
// ════════════════════════════════════════════════════════════════════════
//
// The JS1K shim provides:
//   a  = canvas element
//   b  = document.body
//   c  = canvas 2D rendering context
//   d  = document
//
// The goal is to squeeze a full Pac-Man clone into exactly 1024 bytes by:
//   1. Encoding the maze as binary integers
//   2. Aliasing long canvas method names to 2-char shortcuts
//   3. Sharing variable names aggressively (M, g, h, k, j, l all serve
//      double or triple duty throughout the code)
//   4. Using regpack string compression on the final output
// ════════════════════════════════════════════════════════════════════════

// ── STEP 1: MAZE ENCODING ────────────────────────────────────────────────────
//
// Each row of the Pac-Man maze is stored as a decimal integer.
// When converted to 13-bit binary, each bit represents one cell:
//   1 = wall (blue tile), 0 = open corridor (can walk / has food)
//
// Only the LEFT half (13 columns) is stored. The right half is derived
// by mirroring — Pac-Man's maze is horizontally symmetric.
//
// Examples:
//   8190 → 1111111111110 → top row, solid wall except rightmost
//   4226 → 1000010000010 → row with two wall sections
//   130  → 0000010000010 → row with a single center wall
//
// The variables g, h, k, j, l are reused here to reduce repetition
// (they get overwritten later for other purposes).

M = [
  g = 8190, h = 4226, h, h, g + 1, l = 4240, l,
  l = 8094, 130, 130, k = 159, j = 144, j, 8176,
  j, j, k, j, j, g, h, h, 7423, j = 1168,
  j, l, k = 4098, k, g + 1, f = t = 0   // f=facing offset, t=frame counter
];

C = [];   // C[row][col] = 1 if food present, 0 if eaten. Populated in N(1).

// ── STEP 2: CANVAS METHOD ALIASING ──────────────────────────────────────────
//
// The canvas context 'c' has methods like fillRect, beginPath, translate, etc.
// Each method name is 6–12 characters. In 1K, every character counts.
//
// This loop iterates over every key on 'c' and stores a reference under a
// 2-character alias formed from: c[ methodName[0] + methodName[6] ]
//
// Key mappings that get used later:
//   fillRect    → c["fl"] ... but the code calls c.fc → "fillRec"[0]+"fillRec"[6] = "ft"?
//   Actually: c["fc"] = c.fillRect  (f=fill, c=Rect 7th char... "fillRect"[6]='c' ✓)
//   c["ta"] = c.translate  ("translate"[0]='t', "translate"[6]='a' → 'ta' ✓)
//   c["ba"] = c.beginPath  ("beginPath"[0]='b', "beginPath"[6]='a' → 'ba' ✓... 
//             wait: b-e-g-i-n-P-a → index 6 = 'a' ✓)
//   c["fx"] = c.strokeText ("strokeText"[0]='s'... hmm, "fillText"[0]='f','fillText'[6]='x' ✓)
//
// This is a classic JS1K / code-golf trick to save bytes on repeated API calls.

for (i in c) c[i[0] + i[6]] = c[i];

// ── STEP 3: EXPAND MAP ROWS INTO CELL ARRAYS ────────────────────────────────
//
// Each integer in M is converted to a 13-char binary string (left half only).
// Then the right half is appended by iterating through the string in reverse,
// skipping the center column (j*2 offset ensures true mirror symmetry).
//
// After this loop, M[row] is a 26-character array of '0' and '1' strings,
// and C[row] is an empty array ready to hold food state.

for (i in M) {
  m = M[i].toString(2).padStart(13, 0);   // e.g. "1111111111110"
  for (j in m) m += m[m.length - j * 2 - 1];   // mirror: append chars from right-to-left
  M[i] = m.split('');   // split into individual cell characters
  C[i] = [];
}

// ── HELPER: DRAW A FILLED RECTANGLE ─────────────────────────────────────────
// F(x, y, w, h, colorHex) — sets fillStyle and calls the aliased fillRect.
// Colors are passed as 3-char hex strings (e.g. '009' = #000099 = dark blue,
// '111' = #111111 = dark grey for walls).

F = (x, y, w, h, p) => {
  c.fillStyle = '#' + p;
  c.fc(x, y, w, h, p);
};

// ── STEP 4: DRAW MAP (and optionally reset the game) ─────────────────────────
//
// N() — called every frame to redraw the maze walls
// N(1) — called on game start / game over: also resets food, spawns entities
//
// Wall tiles are 30×30 pixels at 16px grid spacing with an 8px offset,
// giving a 16px grid where walls fill nearly the full cell.

N = (r) => {
  F(0, 0, l, l, '009');   // clear screen: fill entire canvas dark blue

  for (i in M) {
    for (j in M[i]) {

      // Draw a wall tile wherever M[i][j] is '1'
      // +M[i][j] coerces '1'→1 (truthy) and '0'→0 (falsy)
      +M[i][j] && F(8 + j * 16, 8 + i * 16, 30, 30, 111);

      if (r) {
        // Reset food: open cells get food (C=1), walls stay empty (C=0)
        C[i][j] = +M[i][j];   // food = 1 on walls?? Actually walls are drawn on top,
                               // so food only shows in open corridors visually.
                               // The logic inverts: food is eaten from non-wall cells.

        // Factory function for creating a ghost or pacman entity.
        // Each entity gets position (x,y), target tile (z,w), temp turn tile (Z,W),
        // a direction d (1=right 2=down 3=left 4=up), and step counter s.
        // t++ staggers starting positions so ghosts begin with different d values.
        g = (h) => ({
          x: h, y: h,   // current grid tile (integer)
          z: h, w: h,   // next grid tile being moved toward
          Z: h, W: h,   // tile to test when turning
          d: (t++ % 4) + 1,  // initial direction 1–4
          s: 9           // step counter (0–10, triggers tile advance when >9)
        });

        // Spawn 4 ghosts + Pac-Man. Argument is starting grid column.
        // G[0]–G[3] are ghosts, G[4] is Pac-Man.
        G = [g(25), g(5), g(0), g(17), g(16)];
        k = 4;   // player's desired direction (4 = up)
      }
    }
  }
};

N(1);   // Initial draw + full reset

c.lineWidth = 12;   // used for Pac-Man arc stroke width

// Keyboard handler: maps arrow key codes to direction values 1–4.
// Arrow keys: Left=37, Up=38, Right=39, Down=40
//   41 - 37 = 4 (up),  41 - 38 = 3 (left... wait actually:
//   k=41-37=4, k=41-38=3, k=41-39=2, k=41-40=1
// So: Left→4, Up→3, Right→2, Down→1
// These map to d values used in movement logic below.
onkeydown = (e) => k = 41 - e.which;

// ── STEP 5: MAIN GAME LOOP (22ms ≈ 45fps) ───────────────────────────────────

setInterval(() => {

  N();   // Redraw maze background every frame

  // Draw food dots and count remaining food.
  // S starts at 298 (total pellets) and is decremented for each drawn dot.
  S = 298;
  for (i in M) {
    for (j in M[i]) {
      // If food present, draw a 3×3 pixel dot centred in the cell
      C[i][j] && F(22 + j * 16, 22 + i * 16, 3, 3, 'eee', S--);
    }
  }

  c.fx(S, 220, 220);   // Draw score: S = remaining food count (counts down toward 0)
                        // c.fx = c.fillText (aliased). At 0, all food eaten = win.

  // ── ENTITY LOOP: ghosts (i=0–3) then Pac-Man (i=4) ──────────────────────
  for (i in G) {
    with (G[i]) {   // 'with' exposes entity properties (x,y,z,w,d,s,Z,W) directly

      // ── MOVEMENT: advance to next tile every 10 steps ──────────────────────
      // s counts sub-steps 0–9. At s>9, the entity has finished crossing a tile.
      // x,y = tile we CAME FROM (integer grid coords)
      // z,w = tile we are HEADING TO (integer grid coords)
      // Interpolation: rendered position = x + (z-x)*s/10 (smooth glide)
      if (s > 9) {
        x = z;   // commit to new tile
        y = w;
        // Advance z or w based on direction:
        //   d=1: z+=1 (right),  d=2: w+=1 (down)
        //   d=3: z-=1 (left),   d=4: w-=1 (up)
        // d%2 distinguishes horizontal (d=1,3) from vertical (d=2,4)
        z -= d % 2 ? 0 : d - 3;   // d=1→z+=1, d=3→z-=1, d=2,4→no x change
        w -= d % 2 ? d - 2 : 0;   // d=2→w+=1, d=4→w-=1, d=1,3→no y change
        s = 0;
      }

      // ── TURNING: Pac-Man tries to match player input k ──────────────────────
      // Only applies to Pac-Man (i==4) when desired direction k differs from
      // current direction d. We check if the cell in the desired direction is
      // open (M[W][Z] === '1' means it's a wall, so we need +M[W][Z] truthy).
      // Wait — open corridors are '0' in M, walls are '1'.
      // Actually pac-man moves through '0' (open) cells.
      // +M[W][Z] being truthy means it IS a wall — can't turn there.
      // So the turn only fires if +M[W][Z] is truthy... let me re-read:
      //   if (+M[W][Z]) { d=k } — turns IF the test cell is a wall?
      // This appears inverted. In practice, since walls ARE the '1' cells
      // and the maze is built so corridors are narrow, the turn logic relies
      // on the existing position snapping. The original source notes this is
      // "more tricky than expected."
      if (i == 4 && d != k) {
        Z = x - (k % 2 ? 0 : k - 3);   // test tile x in direction k
        W = y - (k % 2 ? k - 2 : 0);   // test tile y in direction k
        if (+M[W][Z]) {
          d = k; z = Z; w = W; s = 0;
        }
      }

      // ── COLLISION / WALL BOUNCE ──────────────────────────────────────────────
      // If the target tile (z,w) is out of bounds or is a wall ('0' open... 
      // again: !+M[w][z] means M[w][z] is '0' which is an open cell — 
      // but open cells are where you CAN move. Walls are '1'.
      // !+M[w][z] fires when the cell is '0' (open). This seems backwards but
      // it works because in this encoding walls block and the entity should
      // bounce back when it tries to enter a '0' (floor) cell that somehow
      // violates bounds — likely the outer boundary rows/cols are '1' walls.
      // In practice: bounce triggers when hitting the maze boundary.
      if (w < 0 || !+M[w][z]) {
        z = x; w = y;   // snap back to previous tile
        // Ghost AI: pick a new direction.
        // (d + t) % 4 + 1 uses the global frame counter t to introduce
        // pseudo-randomness. Each ghost has a different d, so they diverge.
        d = i == 4 ? k : (d + t) % 4 + 1;
        s = 7;   // short pause before trying again
      }

      // ── DRAW GHOSTS (i = 0, 1, 2, 3) ────────────────────────────────────────
      if (i < 4) {
        // Game over: reset if ghost reaches Pac-Man's tile
        if (z == G[4].x && w == G[4].y) N(1);

        // Translate canvas to ghost's interpolated pixel position.
        // The rendered position smoothly glides from tile x to tile z
        // over 10 sub-steps: pixel = 12 + (x + (z-x)*s/10) * 16
        c.ta(
          X = 12 + (x + (z - x) * s / 10) * 16,
          Y = 10 + (y + (w - y) * s / 10) * 16
        );

        // Each ghost has a unique colour. Array indexed by ghost number.
        h = ['f77', 'c70', 'd22', '09c'][i];

        // Draw the ghost body using 6 filled rectangles (relative to translate origin):
        //   Body block, head arc block, leg gap, eye whites, pupils, center stripe
        F(1,  5,  22, 21, h);     // main body
        F(4,  2,  16, 8,  h);     // upper head / dome
        F(6,  23, 13, 3,  111);   // gap between legs (dark background colour)
        F(4,  8,  16, 7,  'FFF'); // eye whites
        F(7,  8,  11, 4,  111);   // dark pupils
        F(10, 0,  4,  26, h);     // top of head + eye divider + middle leg

        c.ta(-X, -Y);   // undo translate
      }

      // ── DRAW PAC-MAN (i = 4) ────────────────────────────────────────────────
      if (i == 4) {
        // Eat: clear food at current integer tile position
        C[y][x] = 0;

        // f = rotation offset so Pac-Man faces his direction of travel.
        // d=1→f=0.5, d=2→f=0, d=3→f=1.5, d=4→f=1 (multiples of π/2)
        f = d % 2 ? d / 2 : d % 3;

        // Mouth animation: t++ increments the global frame counter.
        // (t/6)%2 oscillates 0→1→0→1... slowly (every 6 frames).
        // Subtracting 1.1 and taking abs() gives a triangle wave: mouth
        // opens from ~0.1 up to ~0.9 radians and closes again.
        m = (t++ / 6) % 2 - 1.1;
        if (m < 0) m = -m;   // manual Math.abs() to save bytes

        c.strokeStyle = '#ff0';   // yellow
        c.ba();   // beginPath
        c.arc(
          22 + (x + (z - x) * s / 10) * 16,   // interpolated pixel x
          22 + (y + (w - y) * s / 10) * 16,   // interpolated pixel y
          5,                                   // radius (pixels, small due to scale)
          3.1 * f + m,                         // start angle of arc gap
          3.1 * f - m                          // end angle of arc gap (mouth opening)
        );
        c.stroke();
      }

      s++;   // increment sub-step counter for smooth movement interpolation
    }
  }

}, 22);   // 22ms interval ≈ 45fps
